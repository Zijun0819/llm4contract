```python
import numpy as np
import pandas as pd
from scipy.optimize import linprog
from sklearn.cluster import KMeans

def agent_solver_v2(v: np.ndarray, content: list[dict]) -> np.ndarray:
    """
    Infer a valid agent setting matrix (actions x [probabilities over 5 outcomes + cost])
    that explains all historical interactions under IR and IC constraints.

    Args:
        v: np.ndarray of shape (5,), principal's reward vector for 5 outcomes.
        content: list of dicts, each with keys 'Contract' (list of 5 float payments),
                 'Principal Utility' (float), and 'Agent Action' (1 or -1).

    Returns:
        np.ndarray of shape (n_actions, 6), rows are [p1,...,p5,cost]
    """
    m = v.shape[0]  # number of outcomes (expected 5)
    logs_df = pd.DataFrame(content)

    # Separate accepted and rejected logs
    accept_df = logs_df[logs_df['Agent Action'] == 1]
    reject_df = logs_df[logs_df['Agent Action'] == -1]

    # If no accepted logs, return trivial uniform distribution with zero cost
    if len(accept_df) == 0:
        p_uniform = np.ones(m) / m
        return np.array([np.append(p_uniform, 0.0)])

    accept_contracts = np.vstack(accept_df['Contract'].to_numpy())
    accept_putils = accept_df['Principal Utility'].to_numpy()
    reject_contracts = np.vstack(reject_df['Contract'].to_numpy()) if len(reject_df) > 0 else np.empty((0, m))

    # Step 1: For each accepted contract, infer a plausible outcome distribution p (prob vector)
    # Solve LP for each accepted contract:
    #   sum p = 1
    #   p @ (v - w) = principal utility
    #   0 <= p <= 1
    # Use small tolerance margin for numerical stability

    def solve_p_for_accept(w: np.ndarray, pu: float) -> np.ndarray:
        A_eq = np.vstack([np.ones(m), v - w])
        b_eq = np.array([1.0, pu])
        bounds = [(0, 1) for _ in range(m)]
        res = linprog(c=np.zeros(m), A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        if res.success:
            p = res.x
            p = np.clip(p, 0, None)
            s = p.sum()
            if s > 0:
                p /= s
            else:
                p = np.ones(m) / m
            return p
        else:
            # fallback uniform distribution if infeasible
            return np.ones(m) / m

    candidate_ps_accept = np.array([solve_p_for_accept(w, pu) for w, pu in zip(accept_contracts, accept_putils)])

    # Step 1b: For rejected contracts, infer plausible outcome distributions p_r
    # We do not have principal utility, but agent utility is < 0,
    # so we try to find p_r s.t. sum p_r =1 and p_r @ (w_r) < cost (unknown) < 0,
    # but since cost unknown, we just try to represent rejected contracts by their contracts themselves,
    # as a proxy for clustering, we use normalized contracts as features (payments vector normalized).

    # Normalize rejected contracts rows sum to 1 for clustering proxy
    if reject_contracts.shape[0] > 0:
        rej_norm = reject_contracts - reject_contracts.min(axis=1, keepdims=True)
        rej_norm_sum = rej_norm.sum(axis=1, keepdims=True)
        rej_norm_sum[rej_norm_sum == 0] = 1.0
        candidate_ps_reject = rej_norm / rej_norm_sum
    else:
        candidate_ps_reject = np.empty((0, m))

    # Combine accepted candidate ps and rejected proxy ps for clustering
    candidate_ps_all = np.vstack([candidate_ps_accept, candidate_ps_reject])

    # Step 2: Cluster candidate_ps_all to find representative actions (outcome distributions)
    # Use KMeans with inertia elbow heuristic to adaptively select number of clusters
    max_clusters = min(10, len(candidate_ps_all))
    if max_clusters == 1:
        n_actions = 1
        p_centers = candidate_ps_all.mean(axis=0, keepdims=True)
        accept_assign = np.zeros(len(candidate_ps_accept), dtype=int)
        reject_assign = np.zeros(len(candidate_ps_reject), dtype=int)
    else:
        inertias = []
        km_models = []
        for k in range(1, max_clusters + 1):
            kmeans = KMeans(n_clusters=k, random_state=0, n_init=20).fit(candidate_ps_all)
            inertias.append(kmeans.inertia_)
            km_models.append(kmeans)
        inertias = np.array(inertias)
        deltas = -np.diff(inertias)  # positive reductions
        if len(deltas) >= 2:
            ratios = deltas[1:] / (deltas[:-1] + 1e-12)
            elbow_candidates = np.where(ratios < 0.5)[0] + 2  # +2 because ratios index starts at 2 clusters
            n_actions = elbow_candidates[0] if len(elbow_candidates) > 0 else max_clusters
        elif len(deltas) == 1:
            n_actions = 1 if deltas[0] < 1e-3 else 2
        else:
            n_actions = 1
        kmeans = km_models[n_actions - 1]
        p_centers = kmeans.cluster_centers_
        labels_all = kmeans.labels_
        accept_assign = labels_all[:len(candidate_ps_accept)]
        reject_assign = labels_all[len(candidate_ps_accept):]

    # Step 3: Remove empty clusters robustly (clusters with zero assigned accepted points)
    unique_clusters, counts = np.unique(accept_assign, return_counts=True)
    nonempty_clusters = unique_clusters[counts > 0]
    if len(nonempty_clusters) < n_actions:
        # Keep only clusters with accepted points assigned
        p_centers = p_centers[nonempty_clusters]
        n_actions = len(nonempty_clusters)
        mapping = {old: new for new, old in enumerate(nonempty_clusters)}
        accept_assign = np.array([mapping[a] for a in accept_assign])
        # For rejected assign, remove those assigned to removed clusters by mapping or assign to closest
        # Here we assign rejected to closest remaining cluster by euclidean distance
        if len(reject_assign) > 0:
            rej_ps = candidate_ps_reject
            # Compute distances to remaining clusters
            dists = np.linalg.norm(rej_ps[:, None, :] - p_centers[None, :, :], axis=2)
            reject_assign = np.argmin(dists, axis=1)
    else:
        # No empty clusters, keep reject_assign as is
        pass

    # Step 4: Compute costs from IR constraints (agent rationality on accepted contracts)
    # For each action a:
    #   cost_a <= min_i (p_a @ w_i) for accepted contracts i assigned to a
    #   cost_a >= 0
    ir_margin = 1e-9
    costs = np.zeros(n_actions)
    for a in range(n_actions):
        idxs = np.where(accept_assign == a)[0]
        if len(idxs) == 0:
            costs[a] = 0.0
            continue
        p_a = p_centers[a]
        payoffs = accept_contracts[idxs] @ p_a  # shape (#assigned,)
        costs[a] = max(0.0, np.min(payoffs) - ir_margin)

    # Step 5: Enforce IC constraints from rejected contracts:
    # For each rejected contract w_r and action a:
    #   agent utility < 0 => p_a @ w_r - cost_a < 0 => cost_a > p_a @ w_r
    # So cost_a > max_r p_a @ w_r for all rejected contracts r assigned to any cluster
    # Add a margin proportional to scale for strict inequality
    if reject_contracts.shape[0] > 0:
        rej_utils = reject_contracts @ p_centers.T  # shape (#rej, n_actions)
        # For each action, max over rejected contracts assigned to that action
        max_rej_util = np.full(n_actions, -np.inf)
        for a in range(n_actions):
            idxs = np.where(reject_assign == a)[0]
            if len(idxs) > 0:
                max_rej_util[a] = np.max(rej_utils[idxs, a])
            else:
                max_rej_util[a] = -np.inf
        # For actions with no rejected assigned, consider all rejected contracts to be conservative
        no_rej_assigned = (max_rej_util == -np.inf)
        if np.any(no_rej_assigned):
            # For those actions, use max over all rejected contracts
            global_max = np.max(rej_utils, axis=0)  # max per action over all rejected
            max_rej_util[no_rej_assigned] = global_max[no_rej_assigned]

        margin = np.maximum(1e-7, 1e-4 * np.maximum(1.0, np.abs(max_rej_util)))
        costs = np.maximum(costs, max_rej_util + margin)

    # Numerical safety: ensure costs non-negative
    costs = np.maximum(costs, 0.0)

    # Step 6: Normalize p_centers rows to sum to 1 (numerical safety)
    p_centers = np.clip(p_centers, 0, None)
    row_sums = p_centers.sum(axis=1, keepdims=True)
    row_sums[row_sums == 0] = 1.0
    p_centers /= row_sums

    # Step 7: Final check for IR and IC strictness; adjust costs slightly if violated

    # IR: For accepted contracts i assigned to a, p_a @ w_i - cost_a >= 0
    for a in range(n_actions):
        idxs = np.where(accept_assign == a)[0]
        if len(idxs) == 0:
            continue
        p_a = p_centers[a]
        payoffs = accept_contracts[idxs] @ p_a
        min_agent_util = np.min(payoffs - costs[a])
        if min_agent_util < -1e-10:
            costs[a] = max(0.0, costs[a] + min_agent_util - 1e-10)

    # IC: For rejected contracts r and all actions a, p_a @ w_r - cost_a < 0
    if reject_contracts.shape[0] > 0:
        rej_utils = reject_contracts @ p_centers.T
        for a in range(n_actions):
            max_util = np.max(rej_utils[:, a] - costs[a])
            if max_util >= -1e-10:
                costs[a] = max(costs[a], np.max(rej_utils[:, a]) + 1e-6)

    # Numerical safety again
    costs = np.maximum(costs, 0.0)

    # Step 8: Assemble agent setting matrix: [p1,...,p5, cost]
    agent_setting = np.hstack([p_centers, costs.reshape(-1, 1)])

    return agent_setting
```
