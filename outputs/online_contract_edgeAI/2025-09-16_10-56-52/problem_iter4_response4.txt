```python
import numpy as np
import pandas as pd
from scipy.optimize import linprog
from sklearn.mixture import GaussianMixture

def agent_solver(v: np.ndarray, content: pd.DataFrame) -> np.ndarray:
    n_outcomes = 12
    if len(v) != n_outcomes or content.empty:
        return np.array([[1/n_outcomes] * n_outcomes + [0.0]])
    
    accepted = content[content['Agent Action'] == 1]
    rejected = content[content['Agent Action'] == -1]
    
    def solve_min_lp(w, u_actual):
        c = -w
        A_eq = np.vstack([np.ones(n_outcomes), v - w])
        b_eq = np.array([1.0, u_actual])
        bounds = [(0, 1)] * n_outcomes
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        if res.success:
            return res.x
        return None
    
    candidate_p = []
    for idx, row in accepted.iterrows():
        w = np.array(row['Contract'])
        u_actual = row['Principal Utility']
        p_sol = solve_min_lp(w, u_actual)
        if p_sol is not None:
            candidate_p.append(p_sol)
    
    if not candidate_p:
        candidate_p = [np.ones(n_outcomes) / n_outcomes]
    
    X = np.array(candidate_p)
    n_components_range = range(1, min(9, len(X) + 1))
    best_bic = np.inf
    best_gmm = None
    for n_components in n_components_range:
        gmm = GaussianMixture(n_components=n_components, random_state=0)
        gmm.fit(X)
        bic = gmm.bic(X)
        if bic < best_bic:
            best_bic = bic
            best_gmm = gmm
    
    action_centers = best_gmm.means_
    n_actions = action_centers.shape[0]
    
    def compute_ir_cost(action_idx, contracts):
        if len(contracts) == 0:
            return 0.0
        ws = np.array(contracts)
        p = action_centers[action_idx]
        utilities = p @ ws.T
        return np.min(utilities)
    
    def compute_ic_constraint(action_idx, rejected_contracts):
        if len(rejected_contracts) == 0:
            return -np.inf
        ws = np.array(rejected_contracts)
        p = action_centers[action_idx]
        utilities = p @ ws.T
        return np.max(utilities)
    
    contracts_by_action = [[] for _ in range(n_actions)]
    for idx, row in accepted.iterrows():
        w = np.array(row['Contract'])
        utilities = action_centers @ w
        best_action = np.argmax(utilities)
        contracts_by_action[best_action].append(w)
    
    rejected_contracts = [np.array(row['Contract']) for _, row in rejected.iterrows()]
    
    ir_costs = np.array([compute_ir_cost(a, contracts_by_action[a]) for a in range(n_actions)])
    ic_thresholds = np.array([compute_ic_constraint(a, rejected_contracts) for a in range(n_actions)])
    
    final_costs = np.maximum(ir_costs, ic_thresholds + 1e-8)
    final_costs = np.maximum(final_costs, 0.0)
    
    agent_setting = np.hstack([action_centers, final_costs.reshape(-1, 1)])
    return agent_setting
```
