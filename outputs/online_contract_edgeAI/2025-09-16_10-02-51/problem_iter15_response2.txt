import numpy as np
from scipy.optimize import minimize, Bounds, LinearConstraint, NonlinearConstraint
from sklearn.mixture import GaussianMixture
from sklearn.cluster import KMeans
from scipy.spatial.distance import cdist

def agent_solver_v2(v: np.ndarray, content: list) -> np.ndarray:
    m_outcomes = len(v)
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    if not content:
        return np.zeros((1, m_outcomes + 1))
    
    accepted_contracts = np.array([log['Contract'] for log in accepted_logs])
    accepted_utils = np.array([log['Principal Utility'] for log in accepted_logs])
    
    p_list = []
    for i, w in enumerate(accepted_contracts):
        u_p = accepted_utils[i]
        c = -w
        A_eq = np.vstack([np.ones(m_outcomes), v])
        b_eq = np.array([1, u_p])
        bounds = [(0, 1)] * m_outcomes
        try:
            res = minimize(lambda p: np.dot(c, p), 
                          x0=np.ones(m_outcomes)/m_outcomes,
                          constraints=[{'type': 'eq', 'fun': lambda p: np.sum(p) - 1},
                                      {'type': 'eq', 'fun': lambda p: np.dot(v, p) - u_p}],
                          bounds=bounds,
                          method='trust-constr')
            if res.success:
                p_list.append(res.x)
        except:
            continue
    
    if not p_list:
        p_list = [np.ones(m_outcomes) / m_outcomes]
    
    p_array = np.array(p_list)
    n_candidates = min(12, len(p_list))
    
    if n_candidates > 1:
        kmeans = KMeans(n_clusters=n_candidates, random_state=42, n_init=10)
        labels = kmeans.fit_predict(p_array)
        p_centroids = kmeans.cluster_centers_
        p_centroids = np.clip(p_centroids, 0, 1)
        p_centroids /= p_centroids.sum(axis=1, keepdims=True)
    else:
        p_centroids = np.ones((1, m_outcomes)) / m_outcomes
    
    n_actions = len(p_centroids)
    
    cost_lower_bounds = np.zeros(n_actions)
    epsilon = 1e-8
    for a in range(n_actions):
        p_a = p_centroids[a]
        min_w = np.inf
        for log in accepted_logs:
            w = np.array(log['Contract'])
            agent_util = np.dot(p_a, w)
            if agent_util >= 0:
                min_w = min(min_w, agent_util)
        cost_lower_bounds[a] = max(0, min_w - epsilon) if min_w != np.inf else 0.0
    
    def ic_constraint(costs):
        violations = []
        for i in range(n_actions):
            for j in range(n_actions):
                if i != j:
                    for log in content:
                        w = np.array(log['Contract'])
                        util_i = np.dot(p_centroids[i], w) - costs[i]
                        util_j = np.dot(p_centroids[j], w) - costs[j]
                        violations.append(util_i - util_j)
        return np.array(violations)
    
    def ir_constraint(costs):
        violations = []
        for log in content:
            if log['Agent Action'] == 1:
                w = np.array(log['Contract'])
                max_util = max(np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions))
                violations.append(max_util)
        return np.array(violations)
    
    def objective(costs):
        penalty = 0.0
        penalty_weight = 100.0
        
        ic_violations = ic_constraint(costs)
        penalty += penalty_weight * np.sum(np.maximum(0, -ic_violations))
        
        ir_violations = ir_constraint(costs)
        penalty += penalty_weight * np.sum(np.maximum(0, -ir_violations))
        
        for log in content:
            w = np.array(log['Contract'])
            utils = [np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions)]
            max_util = max(utils)
            if log['Agent Action'] == 1:
                if max_util < -epsilon:
                    penalty += penalty_weight * (-max_util)
            else:
                if max_util >= -epsilon:
                    penalty += penalty_weight * max_util
        
        return penalty
    
    bounds = Bounds(cost_lower_bounds, np.inf)
    
    initial_costs = cost_lower_bounds.copy()
    best_costs = None
    best_obj = np.inf
    
    for init in [initial_costs, np.zeros(n_actions), np.ones(n_actions) * 0.001]:
        res = minimize(objective, init, method='trust-constr', bounds=bounds, options={'maxiter': 1000, 'verbose': 0})
        if res.success and res.fun < best_obj:
            best_obj = res.fun
            best_costs = res.x
    
    if best_costs is None:
        best_costs = initial_costs
    
    agent_setting = np.hstack([p_centroids, best_costs.reshape(-1, 1)])
    return agent_setting
