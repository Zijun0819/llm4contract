```python
import numpy as np
from scipy.optimize import linprog, minimize, Bounds
from sklearn.mixture import BayesianGaussianMixture

def agent_solver_v2(v: np.ndarray, content: list) -> np.ndarray:
    m_outcomes = len(v)
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    if not accepted_logs and not rejected_logs:
        return np.zeros((1, m_outcomes + 1))
    
    p_list = []
    if accepted_logs:
        accepted_contracts = np.array([log['Contract'] for log in accepted_logs])
        accepted_utils = np.array([log['Principal Utility'] for log in accepted_logs])
        
        for i, w in enumerate(accepted_contracts):
            u_p = accepted_utils[i]
            c = -w
            A_eq = np.vstack([np.ones(m_outcomes), v])
            b_eq = np.array([1, u_p])
            bounds = [(0, 1)] * m_outcomes
            res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
            if res.success:
                p_list.append(res.x)
    
    if not p_list and not rejected_logs:
        return np.zeros((1, m_outcomes + 1))
    
    if not p_list:
        p_list = [np.full(m_outcomes, 1.0 / m_outcomes)]
    
    p_array = np.array(p_list)
    n_candidates = min(len(p_list), 12)
    if n_candidates > 0:
        gmm = BayesianGaussianMixture(n_components=n_candidates, random_state=42, max_iter=1000)
        gmm.fit(p_array)
        p_centroids = gmm.means_
    else:
        p_centroids = np.array([np.full(m_outcomes, 1.0 / m_outcomes)])
    
    n_actions = len(p_centroids)
    
    cost_lower_bounds = np.zeros(n_actions)
    for a in range(n_actions):
        p_a = p_centroids[a]
        min_w = np.inf
        for log in accepted_logs:
            w = np.array(log['Contract'])
            if np.dot(p_a, w) >= 0:
                min_w = min(min_w, np.dot(p_a, w))
        if min_w == np.inf:
            min_w = 0.0
        cost_lower_bounds[a] = min_w
    
    def objective(costs):
        violation = 0.0
        for log in content:
            w = np.array(log['Contract'])
            utils = np.array([np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions)])
            max_util = np.max(utils)
            if log['Agent Action'] == 1:
                if max_util < 0:
                    violation += -max_util
                best_action_util = utils[np.argmax(utils)]
                for a in range(n_actions):
                    if a != np.argmax(utils) and utils[a] > best_action_util - 1e-8:
                        violation += (utils[a] - best_action_util + 1e-8)
            else:
                if max_util >= 0:
                    violation += max_util + 1e-8
        return violation
    
    bounds = Bounds(cost_lower_bounds, np.full(n_actions, np.inf))
    constraints = [{'type': 'ineq', 'fun': lambda x: x - cost_lower_bounds}]
    res = minimize(objective, cost_lower_bounds, method='SLSQP', bounds=bounds, constraints=constraints, options={'maxiter': 1000})
    final_costs = res.x if res.success else cost_lower_bounds
    
    agent_setting = np.hstack([p_centroids, final_costs.reshape(-1, 1)])
    return agent_setting
```
