```python
import numpy as np
from scipy.optimize import linprog, minimize, LinearConstraint, Bounds
from sklearn.mixture import GaussianMixture

def agent_solver(v, content):
    v = np.asarray(v)
    n_outcomes = len(v)
    
    if not content:
        uniform_p = np.ones(n_outcomes) / n_outcomes
        cost_zero = np.array([0.0])
        return np.hstack([uniform_p.reshape(1, -1), cost_zero.reshape(-1, 1)])
    
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    accepted_contracts = np.array([log['Contract'] for log in accepted_logs])
    accepted_utils = np.array([log['Principal Utility'] for log in accepted_logs])
    
    p_list = []
    for i in range(len(accepted_logs)):
        w = accepted_contracts[i]
        u_p = accepted_utils[i]
        c = -w
        A_eq = np.vstack([np.ones(n_outcomes), v])
        b_eq = np.array([1, u_p])
        bounds = [(0, 1)] * n_outcomes
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        if res.success:
            p_list.append(res.x)
    
    if not p_list:
        p_list = [np.ones(n_outcomes) / n_outcomes]
    
    p_array = np.array(p_list)
    n_candidates = min(12, max(1, len(p_list)))
    gmm = GaussianMixture(n_components=n_candidates, random_state=42)
    gmm.fit(p_array)
    p_centroids = gmm.means_
    n_actions = p_centroids.shape[0]
    
    cost_lb = np.zeros(n_actions)
    for a in range(n_actions):
        p_a = p_centroids[a]
        min_util = np.inf
        for log in accepted_logs:
            w = np.array(log['Contract'])
            agent_util = np.dot(p_a, w)
            if agent_util >= 0:
                min_util = min(min_util, agent_util)
        cost_lb[a] = max(0, min_util) if min_util != np.inf else 0.0
    
    constraints_list = []
    for log in content:
        w = np.array(log['Contract'])
        for a in range(n_actions):
            eu_a = np.dot(p_centroids[a], w)
            for ap in range(n_actions):
                if a != ap:
                    eu_ap = np.dot(p_centroids[ap], w)
                    constraint_row = np.zeros(n_actions)
                    constraint_row[a] = 1
                    constraint_row[ap] = -1
                    constraints_list.append((constraint_row, eu_a - eu_ap))
    
    if constraints_list:
        A_ic = np.vstack([c[0] for c in constraints_list])
        b_ic = np.array([c[1] for c in constraints_list])
        ic_constraint = LinearConstraint(A_ic, lb=-np.inf, ub=b_ic)
    else:
        ic_constraint = LinearConstraint(np.eye(n_actions), lb=np.zeros(n_actions), ub=np.inf)
    
    def objective(costs):
        violation = 0.0
        for log in content:
            w = np.array(log['Contract'])
            utils = np.array([np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions)])
            max_util = np.max(utils)
            if log['Agent Action'] == 1:
                violation += max(0, -max_util)
            else:
                violation += max(0, max_util)
        return violation
    
    bounds = Bounds(cost_lb, np.full(n_actions, np.inf))
    res = minimize(objective, cost_lb + 1e-6, method='trust-constr', bounds=bounds, constraints=ic_constraint, options={'maxiter': 1000, 'verbose': 0})
    final_costs = res.x if res.success else cost_lb
    
    agent_setting = np.hstack([p_centroids, final_costs.reshape(-1, 1)])
    return agent_setting
```
