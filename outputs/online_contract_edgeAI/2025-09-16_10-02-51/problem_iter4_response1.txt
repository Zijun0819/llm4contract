import numpy as np
from scipy.optimize import linprog, minimize, Bounds
from sklearn.mixture import BayesianGaussianMixture

def agent_solver_v2(v: np.ndarray, content: list) -> np.ndarray:
    m_outcomes = len(v)
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    if not accepted_logs and not rejected_logs:
        return np.zeros((1, m_outcomes + 1))
    
    accepted_contracts = np.array([log['Contract'] for log in accepted_logs])
    accepted_utils = np.array([log['Principal Utility'] for log in accepted_logs])
    rejected_contracts = np.array([log['Contract'] for log in rejected_logs]) if rejected_logs else np.array([]).reshape(0, m_outcomes)
    
    p_list = []
    for i, w in enumerate(accepted_contracts):
        u_p = accepted_utils[i]
        c = -w
        A_eq = np.vstack([np.ones(m_outcomes), v])
        b_eq = np.array([1.0, u_p])
        bounds = [(0, 1)] * m_outcomes
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        if res.success:
            p_list.append(res.x)
    
    if not p_list:
        p_centroids = np.ones((1, m_outcomes)) / m_outcomes
        n_actions = 1
    else:
        p_array = np.array(p_list)
        if len(p_list) == 1:
            p_centroids = p_array.reshape(1, -1)
            n_actions = 1
        else:
            max_components = min(12, len(p_list))
            gmm = BayesianGaussianMixture(n_components=max_components, random_state=42, max_iter=1000)
            gmm.fit(p_array)
            p_centroids = gmm.means_
            n_actions = len(p_centroids)
    
    cost_lower_bounds = np.zeros(n_actions)
    for a in range(n_actions):
        p_a = p_centroids[a]
        min_util = np.inf
        for log in accepted_logs:
            w = np.array(log['Contract'])
            util = np.dot(p_a, w)
            if util >= 0:
                min_util = min(min_util, util)
        cost_lower_bounds[a] = min_util if min_util != np.inf else 0.0
    
    cost_upper_bounds = np.full(n_actions, np.inf)
    for a in range(n_actions):
        p_a = p_centroids[a]
        for log in rejected_logs:
            w = np.array(log['Contract'])
            util = np.dot(p_a, w)
            if util < cost_upper_bounds[a]:
                cost_upper_bounds[a] = util
    
    for i in range(n_actions):
        if cost_upper_bounds[i] == np.inf:
            cost_upper_bounds[i] = cost_lower_bounds[i] + 1.0
        elif cost_upper_bounds[i] < cost_lower_bounds[i]:
            cost_upper_bounds[i] = cost_lower_bounds[i] + 1e-6
    
    def objective(costs):
        violation = 0.0
        for log in content:
            w = np.array(log['Contract'])
            agent_utils = np.array([np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions)])
            max_agent_util = np.max(agent_utils)
            if log['Agent Action'] == 1:
                if max_agent_util < -1e-8:
                    violation += (-max_agent_util)
            else:
                if max_agent_util >= -1e-8:
                    violation += (max_agent_util + 1e-8)
        return violation
    
    bounds = Bounds(cost_lower_bounds, cost_upper_bounds)
    x0 = (cost_lower_bounds + cost_upper_bounds) / 2.0
    res = minimize(objective, x0, method='SLSQP', bounds=bounds, options={'maxiter': 1000, 'ftol': 1e-8})
    final_costs = res.x if res.success else x0
    
    agent_setting = np.hstack([p_centroids, final_costs.reshape(-1, 1)])
    return agent_setting
