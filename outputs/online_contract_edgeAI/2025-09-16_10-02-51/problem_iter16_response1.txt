import numpy as np
from scipy.optimize import linprog, minimize, LinearConstraint, Bounds
from sklearn.mixture import GaussianMixture

def agent_solver(v: np.ndarray, content: list[dict]) -> np.ndarray:
    v = np.asarray(v)
    m_outcomes = len(v)
    
    if not content:
        return np.zeros((1, m_outcomes + 1))
    
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    accepted_contracts = np.array([log['Contract'] for log in accepted_logs])
    accepted_utils = np.array([log['Principal Utility'] for log in accepted_logs])
    
    p_list = []
    for i in range(len(accepted_logs)):
        w = accepted_contracts[i]
        u_p = accepted_utils[i]
        if u_p <= 1e-12:
            continue
        c = -w
        A_eq = np.vstack([np.ones(m_outcomes), v])
        b_eq = np.array([1, u_p])
        bounds = [(0, 1)] * m_outcomes
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        if res.success:
            p_list.append(res.x)
    
    if not p_list:
        p_centroids = np.ones((1, m_outcomes)) / m_outcomes
    else:
        p_array = np.array(p_list)
        n_candidates = min(12, len(p_list))
        if n_candidates > 0:
            gmm = GaussianMixture(n_components=n_candidates, random_state=42)
            gmm.fit(p_array)
            p_centroids = gmm.means_
        else:
            p_centroids = np.ones((1, m_outcomes)) / m_outcomes
    
    n_actions = len(p_centroids)
    
    cost_lower_bounds = np.zeros(n_actions)
    for a in range(n_actions):
        p_a = p_centroids[a]
        min_agent_util = np.inf
        for log in accepted_logs:
            w = np.array(log['Contract'])
            agent_util = np.dot(p_a, w)
            if agent_util >= 0:
                min_agent_util = min(min_agent_util, agent_util)
        cost_lower_bounds[a] = max(0, min_agent_util - 1e-6) if min_agent_util != np.inf else 0.0
    
    constraints_list = []
    for log in content:
        w = np.array(log['Contract'])
        utils = [np.dot(p_centroids[a], w) for a in range(n_actions)]
        if log['Agent Action'] == 1:
            best_action = np.argmax(utils)
            for a in range(n_actions):
                if a != best_action:
                    constraint_matrix = np.zeros(n_actions)
                    constraint_matrix[best_action] = 1
                    constraint_matrix[a] = -1
                    constraint_value = utils[best_action] - utils[a]
                    constraints_list.append((constraint_matrix, constraint_value))
        else:
            for a in range(n_actions):
                constraint_matrix = np.zeros(n_actions)
                constraint_matrix[a] = 1
                constraint_value = utils[a] - 1e-6
                constraints_list.append((constraint_matrix, constraint_value))
    
    if constraints_list:
        A_constraints = np.vstack([c[0] for c in constraints_list])
        b_constraints = np.array([c[1] for c in constraints_list])
        constraints = LinearConstraint(A_constraints, lb=-np.inf, ub=b_constraints)
    else:
        constraints = LinearConstraint(np.eye(n_actions), lb=cost_lower_bounds, ub=np.inf)
    
    def objective(costs):
        violation = 0.0
        for log in content:
            w = np.array(log['Contract'])
            utils = np.array([np.dot(p_centroids[a], w) - costs[a] for a in range(n_actions)])
            max_util = np.max(utils)
            if log['Agent Action'] == 1:
                violation += max(0, -max_util)
            else:
                violation += max(0, max_util)
        return violation
    
    bounds = Bounds(cost_lower_bounds, np.inf * np.ones(n_actions))
    res = minimize(objective, cost_lower_bounds, method='trust-constr', bounds=bounds, constraints=constraints, options={'maxiter': 10000, 'verbose': 0})
    final_costs = res.x if res.success else cost_lower_bounds
    
    agent_setting = np.hstack([p_centroids, final_costs.reshape(-1, 1)])
    return agent_setting
