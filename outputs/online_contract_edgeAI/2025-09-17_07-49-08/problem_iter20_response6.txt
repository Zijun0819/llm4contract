import numpy as np
from scipy.optimize import linprog, minimize
from sklearn.cluster import KMeans

def agent_solver_v2(v: np.ndarray, content: list[dict]) -> np.ndarray:

    n_candidates = 20
    m_outcomes = v.shape[0]
    L = len(content)

    def expected_utility(p, w):
        return np.dot(p, w)

    def mini_lp_p(w: np.ndarray, u: float) -> np.ndarray | None:
        c = np.zeros(m_outcomes)
        A_eq = np.array([np.ones(m_outcomes)])
        b_eq = np.array([1.0])
        A_ub = np.array([-(v - w)])
        b_ub = np.array([-u])
        bounds = [(0, 1)] * m_outcomes
        res = linprog(c, A_eq=A_eq, b_eq=b_eq, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')
        return res.x if res.success else None

    accepted_contracts = []
    accepted_utilities = []
    rejected_contracts = []

    for log in content:
        if log['Agent Action'] == 1:
            accepted_contracts.append(np.array(log['Contract']))
            accepted_utilities.append(log['Principal Utility'])
        else:
            rejected_contracts.append(np.array(log['Contract']))

    if not accepted_contracts and not rejected_contracts:
        return np.array([[1/m_outcomes] * m_outcomes + [0]])

    candidate_ps = []
    for i in range(len(accepted_contracts)):
        w_i = accepted_contracts[i]
        u_i = accepted_utilities[i]
        p_i = mini_lp_p(w_i, u_i)
        if p_i is not None:
            candidate_ps.append(p_i)

    for w in rejected_contracts:
        p_init = np.array([1/m_outcomes] * m_outcomes)
        def objective(p):
            return -expected_utility(p, w)
        constraints = ({'type': 'eq', 'fun': lambda p: np.sum(p) - 1})
        bounds = [(0, 1)] * m_outcomes
        res = minimize(objective, p_init, method='SLSQP', bounds=bounds, constraints=constraints)
        if res.success:
            candidate_ps.append(res.x)

    if not candidate_ps:
         return np.array([[1/m_outcomes] * m_outcomes + [0]])

    all_p = np.array(candidate_ps)
    
    def init_kmeans_plusplus(X, n_clusters, random_state=None):
        if random_state is not None:
            np.random.seed(random_state)
        centers = [X[np.random.choice(len(X))]]
        for _ in range(1, n_clusters):
            distances = np.array([min([np.linalg.norm(x - c)**2 for c in centers]) for x in X])
            probabilities = distances / distances.sum()
            cumulative_probabilities = probabilities.cumsum()
            r = np.random.rand()
            for j, cumulative_probability in enumerate(cumulative_probabilities):
                if r < cumulative_probability:
                    centers.append(X[j])
                    break
        return np.array(centers)

    n_clusters = min(n_candidates, len(all_p))
    initial_centers = init_kmeans_plusplus(all_p, n_clusters, random_state=0)
    kmeans = KMeans(n_clusters=n_clusters, init=initial_centers, n_init=1, random_state=0)
    kmeans.fit(all_p)
    p0 = kmeans.cluster_centers_

    c_init = np.zeros(p0.shape[0])

    for i in range(p0.shape[0]):
        costs = []
        for w, u in zip(accepted_contracts, accepted_utilities):
            costs.append(expected_utility(p0[i], w) - u)
        if costs:
            c_init[i] = max(0, min(costs))
        else:
            c_init[i] = 0

    c_init = np.maximum(0, c_init)

    agent_setting = np.hstack([p0, c_init[:, np.newaxis]])
    return agent_setting
