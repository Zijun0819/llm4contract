import numpy as np
from scipy.optimize import linprog, minimize
from sklearn.cluster import KMeans


def agent_solver_v2(v: np.ndarray, content: list[dict]) -> np.ndarray:
    n_candidates = 10
    m_outcomes = v.shape[0]
    L = len(content)

    def expected_utility(p, contract):
        return np.dot(p, contract)

    def mini_lp_p(w: np.ndarray, u: float) -> np.ndarray | None:
        A_eq = [np.ones(m_outcomes), v - w]
        b_eq = [1.0, u]
        bounds = [(0, 1)] * m_outcomes
        res = linprog(np.zeros_like(w), A_eq=A_eq, b_eq=b_eq, bounds=bounds, method='highs')
        return res.x if res.success else None

    accepted_contracts = []
    accepted_utilities = []
    rejected_contracts = []
    for log in content:
        if log['Agent Action'] == 1:
            accepted_contracts.append(log['Contract'])
            accepted_utilities.append(log['Principal Utility'])
        else:
            rejected_contracts.append(log['Contract'])

    if not accepted_contracts and not rejected_contracts:
        return np.array([[1/m_outcomes] * m_outcomes + [0]])

    candidate_ps = []
    for i in range(len(accepted_contracts)):
        w_i = accepted_contracts[i]
        u_i = accepted_utilities[i]
        p_i = mini_lp_p(w_i, u_i)
        if p_i is not None:
            candidate_ps.append(p_i)

    if not candidate_ps:
        candidate_ps = [np.array([1/m_outcomes] * m_outcomes)]

    all_p = np.array(candidate_ps)
    kmeans = KMeans(n_clusters=min(n_candidates, len(all_p)), random_state=0, n_init=10, init='k-means++').fit(all_p)
    p0 = kmeans.cluster_centers_

    c_init = np.zeros(p0.shape[0])

    for i in range(p0.shape[0]):
        c_init[i] = 0.0

    def utility_loss(c, p, contract):
        return (expected_utility(p, contract) - c)**2

    def prob_constraint(p):
        return np.sum(p) - 1

    def cost_objective(c, p, accepted_contracts):
        loss = 0
        for contract in accepted_contracts:
            loss += utility_loss(c, p, contract)
        return loss

    agent_setting = []
    for i in range(p0.shape[0]):
        p_init = p0[i]
        bounds = [(0, 1)] * m_outcomes
        constraints = ({'type': 'eq', 'fun': prob_constraint})
        result = minimize(lambda p: cost_objective(0, p, accepted_contracts), p_init, method='SLSQP', bounds=bounds, constraints=constraints)
        p_optimized = result.x
        c_est = 0.0
        for log in content:
            if log['Agent Action'] == 1:
                c_est = max(c_est, expected_utility(p_optimized, log['Contract']))
        agent_setting.append(np.concatenate([p_optimized, [c_est]]))

    return np.array(agent_setting)
