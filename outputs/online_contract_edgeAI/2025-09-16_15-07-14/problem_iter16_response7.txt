```python
import numpy as np
from scipy.optimize import linprog
from sklearn.mixture import GaussianMixture
from scipy.spatial.distance import cdist

def agent_solver(v: np.ndarray, content: list[dict]) -> np.ndarray:
    v = np.array(v)
    m = len(v)
    
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    if not accepted_logs:
        return np.hstack([np.eye(m), np.zeros((m, 1))])
    
    def solve_mini_lp(w, u_principal):
        A_eq = np.vstack([np.ones(m), v - w])
        b_eq = np.array([1.0, u_principal])
        res = linprog(-w, A_eq=A_eq, b_eq=b_eq, bounds=[(0, 1)]*m, method='highs')
        return res.x if res.success else None
    
    ps_candidates = []
    for log in accepted_logs:
        w_arr = np.array(log['Contract'])
        p_candidate = solve_mini_lp(w_arr, log['Principal Utility'])
        if p_candidate is not None:
            ps_candidates.append(p_candidate)
    
    if not ps_candidates:
        return np.hstack([np.eye(m), np.zeros((m, 1))])
    
    all_p = np.array(ps_candidates)
    all_p = np.clip(all_p, 0, 1)
    row_sums = all_p.sum(axis=1, keepdims=True)
    all_p = np.where(row_sums > 0, all_p / row_sums, all_p)
    
    n_components = min(10, len(all_p))
    gmm = GaussianMixture(n_components=n_components, random_state=42, covariance_type='diag')
    gmm.fit(all_p)
    centers = gmm.means_
    assignments = gmm.predict(all_p)
    
    costs = np.zeros(n_components)
    for i in range(n_components):
        cluster_indices = np.where(assignments == i)[0]
        if cluster_indices.size > 0:
            contracts = np.array([accepted_logs[j]['Contract'] for j in cluster_indices])
            utilities = contracts @ centers[i]
            costs[i] = np.min(utilities) - 1e-12
        else:
            costs[i] = 0.0
    
    if rejected_logs:
        rejected_contracts = np.array([log['Contract'] for log in rejected_logs])
        rejected_utilities = rejected_contracts @ centers.T
        max_rejected = np.max(rejected_utilities, axis=0)
        costs = np.maximum(costs, max_rejected + 1e-12)
    
    costs = np.maximum(costs, 0)
    
    for idx, center in enumerate(centers):
        for log in accepted_logs:
            w = np.array(log['Contract'])
            utility = w @ center - costs[idx]
            if utility < -1e-12:
                best_utility = np.max(w @ centers.T - costs)
                if utility < best_utility - 1e-12 and np.argmax(w @ centers.T - costs) == idx:
                    costs[idx] = w @ center + 1e-12
    
    for log in rejected_logs:
        w = np.array(log['Contract'])
        best_utility = np.max(w @ centers.T - costs)
        if best_utility > -1e-12:
            best_idx = np.argmax(w @ centers.T - costs)
            costs[best_idx] = w @ centers[best_idx] + 1e-12
    
    distances = cdist(centers, centers)
    np.fill_diagonal(distances, np.inf)
    merge_threshold = 0.01
    merged = set()
    new_centers = []
    new_costs = []
    
    for i in range(n_components):
        if i in merged:
            continue
        group = [i]
        for j in range(i+1, n_components):
            if distances[i, j] < merge_threshold:
                group.append(j)
                merged.add(j)
        merged_center = np.mean(centers[group], axis=0)
        merged_center = merged_center / np.sum(merged_center)
        merged_cost = np.min(costs[group])
        new_centers.append(merged_center)
        new_costs.append(merged_cost)
    
    if new_centers:
        centers = np.array(new_centers)
        costs = np.array(new_costs)
        n_components = len(centers)
    
    for i in range(n_components):
        for log in accepted_logs:
            w = np.array(log['Contract'])
            utility_i = w @ centers[i] - costs[i]
            max_other = np.max(np.delete(w @ centers.T - costs, i))
            if utility_i < max_other - 1e-12:
                costs[i] = w @ centers[i] - max_other + 1e-12
    
    for log in rejected_logs:
        w = np.array(log['Contract'])
        best_utility = np.max(w @ centers.T - costs)
        if best_utility > -1e-12:
            best_idx = np.argmax(w @ centers.T - costs)
            costs[best_idx] = w @ centers[best_idx] + 1e-12
    
    centers = np.clip(centers, 0, 1)
    row_sums = centers.sum(axis=1, keepdims=True)
    centers = np.where(row_sums > 0, centers / row_sums, centers)
    
    agent_setting = np.hstack([centers, costs.reshape(-1, 1)])
    return agent_setting
```
