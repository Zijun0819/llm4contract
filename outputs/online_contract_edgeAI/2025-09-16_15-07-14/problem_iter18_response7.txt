import numpy as np
from scipy.optimize import linprog
from sklearn.cluster import KMeans
from sklearn.mixture import GaussianMixture
from sklearn.preprocessing import normalize

def agent_solver(v, content):
    v = np.array(v)
    m = len(v)
    
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    
    if not accepted_logs:
        return np.hstack([np.eye(m), np.zeros((m, 1))])
    
    def solve_mini_lp(w, u_principal):
        w_arr = np.array(w)
        A_eq = np.vstack([np.ones(m), v - w_arr])
        b_eq = np.array([1.0, u_principal])
        res = linprog(-w_arr, A_eq=A_eq, b_eq=b_eq, bounds=[(0, 1)] * m, method='highs')
        if res.success:
            p = res.x
            p = np.maximum(p, 0)
            p_sum = p.sum()
            if p_sum > 0:
                return p / p_sum
        return None
    
    ps_candidates = []
    for log in accepted_logs:
        p_candidate = solve_mini_lp(log['Contract'], log['Principal Utility'])
        if p_candidate is not None:
            ps_candidates.append(p_candidate)
    
    if not ps_candidates:
        return np.hstack([np.eye(m), np.zeros((m, 1))])
    
    all_p = np.array(ps_candidates)
    all_p = normalize(all_p, norm='l1', axis=1)
    
    n_clusters_kmeans = min(10, max(2, len(all_p) // 10))
    kmeans = KMeans(n_clusters=n_clusters_kmeans, random_state=42, n_init=10)
    kmeans_labels = kmeans.fit_predict(all_p)
    kmeans_centers = kmeans.cluster_centers_
    kmeans_centers = np.maximum(kmeans_centers, 0)
    kmeans_centers = normalize(kmeans_centers, norm='l1', axis=1)
    
    kmeans_costs = np.zeros(n_clusters_kmeans)
    for i in range(n_clusters_kmeans):
        cluster_indices = np.where(kmeans_labels == i)[0]
        if cluster_indices.size > 0:
            contracts = np.array([accepted_logs[j]['Contract'] for j in cluster_indices])
            utilities = contracts @ kmeans_centers[i]
            kmeans_costs[i] = np.maximum(np.min(utilities) - 1e-12, 0.0)
    
    n_components_gmm = max(2, n_clusters_kmeans // 2)
    gmm = GaussianMixture(n_components=n_components_gmm, random_state=42, covariance_type='diag')
    gmm_labels = gmm.fit_predict(all_p)
    gmm_centers = gmm.means_
    gmm_centers = np.maximum(gmm_centers, 0)
    gmm_centers = normalize(gmm_centers, norm='l1', axis=1)
    
    gmm_costs = np.zeros(n_components_gmm)
    for i in range(n_components_gmm):
        cluster_indices = np.where(gmm_labels == i)[0]
        if cluster_indices.size > 0:
            contracts = np.array([accepted_logs[j]['Contract'] for j in cluster_indices])
            utilities = contracts @ gmm_centers[i]
            gmm_costs[i] = np.maximum(np.min(utilities) - 1e-12, 0.0)
    
    if rejected_logs:
        rejected_contracts = np.array([log['Contract'] for log in rejected_logs])
        kmeans_rejected_utilities = rejected_contracts @ kmeans_centers.T
        kmeans_max_rejected = np.max(kmeans_rejected_utilities, axis=0)
        kmeans_costs = np.maximum(kmeans_costs, kmeans_max_rejected + 1e-12)
        
        gmm_rejected_utilities = rejected_contracts @ gmm_centers.T
        gmm_max_rejected = np.max(gmm_rejected_utilities, axis=0)
        gmm_costs = np.maximum(gmm_costs, gmm_max_rejected + 1e-12)
    
    kmeans_agent_setting = np.hstack([kmeans_centers, kmeans_costs.reshape(-1, 1)])
    gmm_agent_setting = np.hstack([gmm_centers, gmm_costs.reshape(-1, 1)])
    
    return gmm_agent_setting
