```python
import numpy as np
from scipy.optimize import linprog
from sklearn.cluster import KMeans

def agent_solver(v: np.ndarray, content: list) -> np.ndarray:
    accepted_logs = [log for log in content if log['Agent Action'] == 1]
    rejected_logs = [log for log in content if log['Agent Action'] == -1]
    n_outcomes = len(v)
    
    if not accepted_logs:
        return np.hstack([np.eye(n_outcomes), np.zeros((n_outcomes, 1))])
    
    def solve_feasible_p(w, u_principal):
        A_eq = np.vstack([np.ones(n_outcomes), v - w])
        b_eq = np.array([1, u_principal])
        result = linprog(-w, A_eq=A_eq, b_eq=b_eq, bounds=[(0, 1)] * n_outcomes, method='highs')
        if result.success:
            return result.x
        return None
    
    candidate_ps = []
    for log in accepted_logs:
        p = solve_feasible_p(np.array(log['Contract']), log['Principal Utility'])
        if p is not None:
            candidate_ps.append(p)
    
    if not candidate_ps:
        return np.hstack([np.eye(n_outcomes), np.zeros((n_outcomes, 1))])
    
    candidate_ps = np.array(candidate_ps)
    n_clusters = min(30, len(candidate_ps))
    kmeans = KMeans(n_clusters=n_clusters, random_state=42, n_init=10).fit(candidate_ps)
    cluster_centers = kmeans.cluster_centers_
    
    cluster_costs = np.zeros(n_clusters)
    for i in range(n_clusters):
        cluster_indices = np.where(kmeans.labels_ == i)[0]
        if len(cluster_indices) > 0:
            contracts = [np.array(accepted_logs[j]['Contract']) for j in cluster_indices]
            utilities = [np.dot(cluster_centers[i], w) for w in contracts]
            cluster_costs[i] = np.max(utilities)
    
    if rejected_logs:
        rej_contracts = np.array([log['Contract'] for log in rejected_logs])
        rej_utilities = cluster_centers @ rej_contracts.T
        for i in range(n_clusters):
            max_rej_util = np.max(rej_utilities[i])
            if max_rej_util > cluster_costs[i]:
                cluster_costs[i] = max_rej_util
    
    for i in range(n_clusters):
        cluster_costs[i] = max(0, cluster_costs[i])
    
    return np.hstack([cluster_centers, cluster_costs.reshape(-1, 1)])
```
